```javascript
{
    int b
    int a
    
    {
        a = 0
        boolean a
        a = true
        {
            boolean b
            b = false
        }
        {
            a = false
            int a
            a = 9
        }
    }
    
    {
        a = 1
        boolean b
        b = true
    }
    
    b = 1
}$
```
